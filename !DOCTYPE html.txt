const canvas=document.getElementById('game'),ctx=canvas.getContext('2d');
let W=canvas.width,H=canvas.height;

let player={x:120,y:H-60,w:40,h:40,vy:0,gravity:0.9,jump:-14,onGround:true};
let obstacles=[],distance=0,speed=6,running=false,level=0;

const levels=[
{len:2000,obs:[{x:400,w:40,h:40},{x:800,w:50,h:50}]},
{len:3000,obs:[{x:300,w:50,h:50},{x:900,w:60,h:60}]},
{len:4000,obs:[{x:500,w:60,h:60},{x:1200,w:50,h:50}]}
];

function startLevel(idx){
level=idx;running=true;distance=0;player.y=H-60;obstacles=levels[idx].obs.map(o=>({...o,x:o.x}));
document.getElementById('menu').style.display='none';
}

function jump(){if(player.onGround){player.vy=player.jump;player.onGround=false}}

window.addEventListener('keydown',e=>{if(e.code==='Space'||e.code==='ArrowUp')jump()});

function update(){
if(!running)return;
player.vy+=player.gravity;
player.y+=player.vy;
if(player.y+player.h>=H){player.y=H-player.h;player.vy=0;player.onGround=true;}

distance+=speed;
obstacles=obstacles.filter(o=>o.x+o.w>0);
if(distance>=levels[level].len){running=false;alert('Level dokončený!');document.getElementById('menu').style.display='flex'}
}
